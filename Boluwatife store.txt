<script>

let cart = [];
let total = 0;

/* CART SYSTEM */
function addCart(name, price){
  cart.push({name, price});
  total += price;

  renderCart();
}

function renderCart(){
  let c = document.getElementById("cart");
  c.innerHTML = "";

  cart.forEach(item => {
    c.innerHTML += `<p>${item.name} - ₦${item.price}</p>`;
  });

  document.getElementById("total").innerText = "Total: ₦" + total;
}

/* PAYSTACK */
function pay(){
  let handler = PaystackPop.setup({
    key: "pk_test_xxxxxxxxxxxxx", // replace
    email: "customer@email.com",
    amount: total * 100,
    currency: "NGN",

    callback: function(res){
      alert("Payment successful! Ref: " + res.reference);
      cart = [];
      total = 0;
      renderCart();
    }
  });

  handler.openIframe();
}

/* CHATBOT */
function addMsg(text, cls){
  let chat = document.getElementById("chat");
  chat.innerHTML += `<div class="msg ${cls}">${text}</div>`;
}

function send(){
  let input = document.getElementById("input");
  let text = input.value;
  if(!text) return;

  addMsg(text, "user");

  let res = ai(text.toLowerCase());

  setTimeout(() => addMsg(res, "bot"), 400);

  input.value = "";
}

function ai(t){
  if(t.includes("headache")) return "💊 Paracetamol recommended";
  if(t.includes("malaria")) return "💊 Artemether combo recommended";
  if(t.includes("cough")) return "💊 Cough syrup recommended";
  return "⚠️ Please consult a pharmacist for proper diagnosis";
}

/* VOICE */
function voice(){
  let rec = new (window.SpeechRecognition || window.webkitSpeechRecognition)();
  rec.lang = "en-US";
  rec.start();

  rec.onresult = e => {
    document.getElementById("input").value = e.results[0][0].transcript;
    send();
  };
}
